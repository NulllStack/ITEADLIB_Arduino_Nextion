20 /*

35 void loop(void)
36 {
37  /*
38  * when one second passed,update the current passed second to the text attribute of component.
39  */
40  itoa(ntimeCount, szTimeCount, 10);
41  qr0.setText(szTimeCount);
42  ntimeCount++;
43  delay(1000);
44 }
